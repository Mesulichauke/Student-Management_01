function logout() {
        // Redirect to the home page
        window.location.href = 'index.html'; // Change 'index.html' to your actual home page filename

    // Wait for the DOM to be fully loaded
document.addEventListener('DOMContentLoaded', function() {
    // Notification Button
    const notifyBtn = document.getElementById('notify-btn');
    if (notifyBtn) {
        notifyBtn.addEventListener('click', handleNotificationClick);
    }
    function handleNotificationClick() {
    const notificationPanel = document.getElementById('notification-panel');
    const notificationDropdown = document.getElementById('notification-dropdown');

    if (notificationPanel.style.display === 'block') {
        // Close the notification panel
        notificationPanel.style.display = 'none';
        notificationDropdown.style.display = 'none';
    } else {
        // Open the notification panel and refresh notifications
        notificationPanel.style.display = 'block';
        checkForNotifications();
    }
}

    // Close Notification Panel Button
    const closeNotificationBtn = document.getElementById('close-notification-btn');
    if (closeNotificationBtn) {
        closeNotificationBtn.addEventListener('click', toggleNotificationPanel);
    }

    // Dropdown Buttons
    const dropdownButtons = document.querySelectorAll('.dropdown-btn');
    dropdownButtons.forEach(function(button) {
        button.addEventListener('click', function() {
            const targetId = button.getAttribute('data-target');
            toggleDropdown(targetId);
        });
    });

    // Attendance Register Button
    const attendanceRegisterBtn = document.getElementById('attendance-register-btn');
    if (attendanceRegisterBtn) {
        attendanceRegisterBtn.addEventListener('click', toggleAttendanceForm);
    }

    // Submit Attendance Button
    const submitAttendanceBtn = document.getElementById('submit-attendance-btn');
    if (submitAttendanceBtn) {
        submitAttendanceBtn.addEventListener('click', submitAttendance);
    }

    // Attendance Buttons
    document.getElementById('view-attendance-btn').addEventListener('click', toggleAttendanceDropdown);
    document.getElementById('student-attendance-btn').addEventListener('click', function() {
        showAttendance('student');
    });
    document.getElementById('teacher-attendance-btn').addEventListener('click', function() {
        showAttendance('teacher');
    });
    document.getElementById('sgb-attendance-btn').addEventListener('click', function() {
        showAttendance('sgb');
    });
    document.getElementById('attendance-history-btn').addEventListener('click', showAttendanceHistory);

    // Leadership Ratings Button
    document.getElementById('leadership-ratings-btn').addEventListener('click', showLeadershipRatings);

    // Submit Leadership Scores
    document.getElementById('submit-sgb-score-btn').addEventListener('click', function() {
        submitLeadershipScore('sgb');
    });

    // Function to toggle the display of the leadership ratings section
    function toggleLeadershipRatings() {
        const leadershipOutput = document.getElementById('leadership-output');
        
        if (leadershipOutput.style.display === 'none' || leadershipOutput.style.display === '') {
            // Show the leadership ratings content
            leadershipOutput.style.display = 'block';
        } else {
            // Hide the leadership ratings content
            leadershipOutput.style.display = 'none';
        }
    }

    document.getElementById('submit-hod-score-btn').addEventListener('click', function() {
        submitLeadershipScore('hod');
    });

    // Pull Report Button
    document.getElementById('pull-report-btn').addEventListener('click', pullReport);

    // Budget Report Button
    document.getElementById('view-budget-report-btn').addEventListener('click', toggleBudgetReport);

    // Budget Form Submission
    document.getElementById('budgetForm').addEventListener('submit', handleBudgetFormSubmit);
});

// Toggle Dropdown Content
function toggleDropdown(dropdownId) {
    const dropdown = document.getElementById(dropdownId);
    if (dropdown) {
        dropdown.classList.toggle('active');
    }
}

// Toggle Notification Panel and Check for Notifications
function handleNotificationClick() {
    const notificationPanel = document.getElementById('notification-panel');
    const notificationDropdown = document.getElementById('notification-dropdown');

    if (notificationPanel.style.display === 'block') {
        // Hide the notification panel
        notificationPanel.style.display = 'none';
        notificationDropdown.style.display = 'none'; // Hide dropdown when panel is closed
    } else {
        // Show the notification panel and refresh notifications
        notificationPanel.style.display = 'block';
        checkForNotifications();
    }
}

// Toggle Notification Panel
function toggleNotificationPanel() {
    const notificationPanel = document.getElementById('notification-panel');
    if (notificationPanel.style.display === 'block') {
        notificationPanel.style.display = 'none';
    } else {
        notificationPanel.style.display = 'block';
    }
}

// Handle Notification Click
function handleNotificationClick() {
    toggleNotificationPanel();
    checkForNotifications();
}

// Check for Notifications
function checkForNotifications() {
    const dropdown = document.getElementById('notification-dropdown');
    const notificationList = document.getElementById('notification-list');
    notificationList.innerHTML = "";

    const studentPerformance = parseInt(document.getElementById('student-performance').textContent);
    const studentAttendance = parseInt(document.getElementById('student-attendance').textContent);
    const teacherPerformance = parseInt(document.getElementById('teacher-performance').textContent);
    const teacherAttendance = parseInt(document.getElementById('teacher-attendance').textContent);

    let hasIrregularities = false;

    if (studentPerformance < 50) {
        hasIrregularities = true;
        notificationList.appendChild(createNotification("Student A has low performance (below 50%)."));
    }

    if (studentAttendance < 80) {
        hasIrregularities = true;
        notificationList.appendChild(createNotification("Student A has low attendance (below 80%)."));
    }

    if (teacherPerformance < 50) {
        hasIrregularities = true;
        notificationList.appendChild(createNotification("Teacher X has low performance (below 50%)."));
    }

    if (teacherAttendance < 80) {
        hasIrregularities = true;
        notificationList.appendChild(createNotification("Teacher X has low attendance (below 80%)."));
    }

    if (hasIrregularities) {
        dropdown.style.display = 'block';
    } else {
        dropdown.style.display = 'none';
    }
}

// Create Notification List Item
function createNotification(message) {
    const notification = document.createElement("li");
    notification.textContent = message;
    return notification;
}

// Toggle Attendance Form
function toggleAttendanceForm() {
    const attendanceInfo = document.getElementById("attendanceInfo");
    attendanceInfo.style.display = attendanceInfo.style.display === "none" || attendanceInfo.style.display === '' ? "block" : "none";
}

// Submit Attendance
function submitAttendance() {
    const date = document.getElementById("attendanceDate").value;
    const status = document.getElementById("attendanceStatus").value;

    if (date && status) {
        const tbody = document.querySelector("#attendanceHistoryTable tbody");
        const newRow = tbody.insertRow();
        newRow.innerHTML = `<td>${date}</td><td>${status}</td>`;

        // Clear the form fields
        document.getElementById("attendanceDate").value = "";
        document.getElementById("attendanceStatus").value = "";
    } else {
        alert("Please fill in both the date and status.");
    }
}

// Toggle Attendance Dropdown
function toggleAttendanceDropdown() {
    const dropdown = document.getElementById('attendance-dropdown');
    dropdown.style.display = dropdown.style.display === 'none' ? 'block' : 'none';
}

// Submit Leadership Score
function submitLeadershipScore(type) {
    const scoreElement = document.getElementById(`${type}-score`);
    const score = scoreElement.value;
    if (score) {
        alert(`Score for ${type.toUpperCase()}: ${score}`);
        // Reset the selection
        scoreElement.value = "";
    } else {
        alert('Please enter a valid score.');
    }
}

// Pull Report
function pullReport() {
    const selectedProfile = document.getElementById('profile-dropdown').value;
    const output = document.getElementById('report-output');
    output.innerHTML = selectedProfile ? `<p>Pulling report for ${capitalizeFirstLetter(selectedProfile)}...</p>` : "<p>Please select a profile.</p>";
}

// Toggle Budget Report
function toggleBudgetReport() {
    const budgetReport = document.getElementById('budget-report');
    budgetReport.style.display = budgetReport.style.display === 'none' || budgetReport.style.display === '' ? 'block' : 'none';
}

// Handle Budget Form Submission
function handleBudgetFormSubmit(event) {
    event.preventDefault();
    const category = document.getElementById('category').value;
    const amount = document.getElementById('amount').value;
    const description = document.getElementById('description').value;

    if (category && amount && description) {
        const tbody = document.querySelector('#budgetTable tbody');
        const historyTbody = document.getElementById('budgetHistoryTable');
        const currentDate = new Date().toLocaleDateString();

        // Add new budget entry
        const newBudgetRow = tbody.insertRow();
        newBudgetRow.innerHTML = `<td>${category}</td><td>${amount}</td><td>${description}</td>`;

        // Add to history
        const newHistoryRow = historyTbody.insertRow();
        newHistoryRow.innerHTML = `<td>${currentDate}</td><td>${category}</td><td>${amount}</td><td>${description}</td>`;

        // Clear the form fields
        document.getElementById('category').value = '';
        document.getElementById('amount').value = '';
        document.getElementById('description').value = '';
    } else {
        alert('Please fill out all fields.');
    }
}

// Utility Function to Capitalize First Letter
function capitalizeFirstLetter(string) {
    return string.charAt(0).toUpperCase() + string.slice(1);
}
