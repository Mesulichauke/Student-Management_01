function toggleVisibility(id) {

function viewTeacherScoring() {
        // Example scoring logic based on student performance
        const studentPerformances = [
            { name: "Student A", grade: 85 },
            { name: "Student B", grade: 78 },
            { name: "Student C", grade: 92 },
            { name: "Student D", grade: 70 }
        ];

        const totalGrades = studentPerformances.reduce((sum, student) => sum + student.grade, 0);
        const averagePerformance = totalGrades / studentPerformances.length;

        let score = 0;
        if (averagePerformance >= 90) {
            score = "Excellent";
        } else if (averagePerformance >= 75) {
            score = "Good";
        } else if (averagePerformance >= 50) {
            score = "Average";
        } else {
            score = "Needs Improvement";
        }

        const resultDiv = document.getElementById("teacher-scoring-result");
        resultDiv.innerHTML = `<p>Teacher's Score: <strong>${score}</strong></p>`;
        resultDiv.style.display = "block";
    }

    // Function to submit the attendance and update the history
    function submitAttendance() {
        const date = document.getElementById("attendanceDate").value;
        const status = document.getElementById("attendanceStatus").value;
        
        if (!date || !status) {
            alert("Please fill in both date and status.");
            return;
        }

        const table = document.getElementById("attendanceTable").getElementsByTagName('tbody')[0];
        const newRow = table.insertRow(table.rows.length);
        const dateCell = newRow.insertCell(0);
        const statusCell = newRow.insertCell(1);

        dateCell.textContent = date;
        statusCell.textContent = status;

        // Clear form inputs
        document.getElementById("attendanceDate").value = "";
        document.getElementById("attendanceStatus").value = "";
    }

document.getElementById("submit-scores").addEventListener("click", function() {
  // Get the student scores from the input fields
  const studentScores = Array.from(document.querySelectorAll(".student-input"))
    .map(input => parseFloat(input.value))
    .filter(score => !isNaN(score));

  

  // Calculate the average score
  const totalScore = studentScores.reduce((total, score) => total + score, 0);
  const averageScore = totalScore / studentScores.length;

  // Display Teacher's Performance based on the average score
  const performanceScore = calculateTeacherPerformance(averageScore);
  document.getElementById("performance-score").textContent = performanceScore.toFixed(2);
});

// Function to calculate teacher's performance based on student average score
function calculateTeacherPerformance(averageScore) {
  let performance = 0;
  if (averageScore >= 90) {
    performance = 100; // Excellent performance
  } else if (averageScore >= 75) {
    performance = 85; // Good performance
  } else if (averageScore >= 60) {
    performance = 70; // Satisfactory performance
  } else {
    performance = 50; // Needs improvement
  }
  return performance;
}
